* locdog

** Usage

write $HOME/.config/locdog/config.json as below

#+begin_src json
  {
      "default_interval": 5,
      "default_threshold": 300,
      "default_alert_cmd": [
          "echo"
      ],
      "targets": [
          {
              "alert_cmd": [
                  "terminal-notifier",
                  "-title",
                  "LocDog",
                  "-sound",
                  "default",
                  "-message",
                  "you shoud org-clock-in"
              ],
              "watch_cmd": [
                  "bash",
                  "-c",
                  "test $(emacsclient -e '(org-clock-is-active)') = t"
              ]
          }
      ]
  }
#+end_src
